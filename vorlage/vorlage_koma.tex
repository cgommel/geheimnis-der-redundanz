% --- Aufgabentyp-Marker in der Marginalie ---
% Verwendung: \aufgabenmarker{Symbol}{Zeile1}{Zeile2}
% z.B.: \aufgabenmarker{✏}{Bleistift-}{Übung 3}
\newcommand{\aufgabenmarker}[3]{%
  \marginpar{%
    \raggedright\color{markercolor}%
    {\markerfont\fontsize{26}{30}\selectfont #1}\\[0.15em]
    {\small\textsc{#2}}\\
    {\small\textsc{#3}}%
  }%
  \ignorespaces%
}

% Spezifische Wrapper für die drei Aufgabentypen:
\newcommand{\bleistiftuebung}[2]{%
  \aufgabenmarker{✏}{Bleistift-}{Übung #1}\textbf{#2}\par\medskip%
}
\newcommand{\pythoneinheit}[2]{%
  \aufgabenmarker{⌨}{Python-}{Einheit #1}\textbf{#2}\par\medskip%
}
\newcommand{\werkzeugcheck}[1]{%
  \aufgabenmarker{⚙}{Werkzeug-}{Check}\textbf{#1}\par\medskip%
}
